import threading
import time

CRANE_WORK_TIME = {crane.name: 0 for crane in io.crane_io.devices}

def crane_work_time():
    while True:
        for crane in io.crane_io.devices:
            if crane.status:
                CRANE_WORK_TIME[crane.name] += 1
        time.sleep(1)
        
threading.Thread(target=crane_work_time).start()
